let x=250;
let y=250;
let vx=5;
let vy=10;
let r=20;
let bx=0;
let w=10;

描画エリア

x=x+vx;
y=y+vy;
if(x<=r||x>=500-r){
  vx=-1*vx;
}
if(y<=r||y>=500-r){
  vy=-1*vy;
}
setColor("yellow");
if(vy!=0){
  bx=x+vx*(-y/vy);
  if(bx<0){
    drawRect(w/2,250,w,500);
  }
  else if(bx>500){
    drawRect(500-w/2,250,w,500);
  }
  else{
    if(vy<0){
      drawRect(250,w/2,500,w);
    }
    else{
      drawRect(250,500-w/2,500,w);
    }
  }
}
else{
  if(vx<0){
    drawRect(w/2,250,w,500);
  }
  else if(vx>0){
    drawRect(500-w/2,250,w,500);
  }
}
setColor("red");
drawCircle(x,y,r);

課題3

変数定義エリア

let x=250;
let y=250;
let vx=5;
let vy=10;
let r=20;
let pr=1.1;
let mr=200;

描画エリア

x=x+vx;
y=y+vy;
if(y>=500-r&&vy>0){
  vy=-1*vy;
  r=min(mr,r*pr);
}
if(x>=500-r&&vx>0){
  vx=-1*vx;
  r=min(mr,r*pr);
}
if(y<=r&&vy<0){
  vy=-1*vy;
  r=min(mr,r*pr);
}
if(x<=r&&vx<0){
  vx=-1*vx;
  r=min(mr,r*pr);
}
setColor("red");
drawCircle(x,y,r);

課題4

変数定義エリア

let t=0;
let cx=250;
let cy=250;
let cr=60;
let dist=150;
let w=50;
let dx=0;
let dy=0;

描画エリア

dx=mouseX-250;
dy=mouseY-250;
if(dx**2+dy**2<=cr**2){
  t=t+2;
}
else{
  t=t-2;
}
setColor("red");
drawCircle(cx,cy,cr);
setColor("pink");
setRotation(t*2);
drawRect(cx+dist*cos(t),cy+dist*sin(t),w,w);
drawRect(cx-dist*cos(t),cy-dist*sin(t),w,w);

課題5

変数定義エリア

let x=250;
let y=250;
let r=20;
let dx=0,dy=0,d=0;
let div=5;

描画エリア

dx=mouseX-x;
dy=mouseY-y;
d=sqrt(dx**2+dy**2)/div;
if(d>1){
  x=x+dx/d;
  y=y+dy/d;
}
setColor("red");
drawCircle(x,y,r);
